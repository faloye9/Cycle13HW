# Create list of Celsius valuesc = list(range(0, 101, 5))# Create list of Fahrenheit valuesf = []for x in c:	f.append((x*(9/5)+32))# Create value pairs, five in a row, separated by ‘/’, with lines above and belowwhile len(c) and len(f):	pairs = []	n = 0	while n<5:		d = c.pop(0)		e = f.pop(0)		g = str(d) + "/" + str(e) + " |"		pairs.append(g)		n+=1	print ("------------------------------------------------------")	print(" ".join(pairs))
