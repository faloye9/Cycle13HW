validiput = Falsetries = 0triesleft = 3while (validiput==False):	try:		from random import randint		randnum = (randint(1, 10))		guess = int(input("I have an integer between 1 and 10. Guess it:"))		while tries < 2:			if guess == randnum:				print("You got it!")				tries = 3				validiput=True			elif (randnum-guess==1) or (randnum-guess==-1):				tries+=1				triesleft-=1				print("Hot! You have",str(triesleft), "tries left.")				guess=int(input("Try again:"))			elif (randnum-guess==2) or (randnum-guess==-2):				tries+=1				triesleft-=1				print("Warm. You have",str(triesleft), "tries left.")				guess=int(input("Try again:"))			elif (randnum-guess>2) or (randnum-guess<-2):				tries+=1				triesleft-=1				print("Cold. You have",str(triesleft), "tries left.")				guess=int(input("Try again:"))		if (tries==2) and (randnum!=guess):			print("Sorry, the number was", str(randnum)+".")			validiput=True	except ValueError:		tries+=1		triesleft-=1		if (tries>0) and (triesleft>0):			print("Enter an integer, ok? Again...")		else:			print("That wasn't an integer. But you have no tries left. Sorry.")			break
